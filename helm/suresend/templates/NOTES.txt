  kubectl port-forward svc/{{ include "suresend.fullname" . }}-web 3000:{{ .Values.web.service.port }}
  kubectl port-forward svc/{{ include "suresend.fullname" . }}-api 4000:{{ .Values.api.service.port }}
{{- end }}

NOTE: Ensure you have set real values for api.secrets before going to production:
  - api.secrets.jwtSecret
  - api.secrets.databaseUrl
  - api.secrets.redisUrl
  - api.secrets.stripeSecretKey

{{- if .Values.unbound.enabled }}
DNS: A local unbound recursive resolver is deployed ({{ include "suresend.fullname" . }}-unbound).
     The reputation service will automatically use it for accurate RBL blacklist checks
     by resolving the service name at startup (no manual IP configuration required).
{{- else }}
DNS: unbound is disabled (unbound.enabled=false). RBL blacklist checks may show false
     positives if the cluster uses a public upstream resolver (e.g. 8.8.8.8).
     Enable with: --set unbound.enabled=true
{{- end }}
